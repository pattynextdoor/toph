type SessionStatus int

const (
	StatusActive SessionStatus = iota
	StatusWaiting
	StatusIdle
	StatusError
	StatusDead
)

func (s SessionStatus) String() string {
	switch s {
	case StatusActive:
		return "active"
	case StatusWaiting:
		return "waiting"
	case StatusIdle:
		return "idle"
	case StatusError:
		return "error"
	case StatusDead:
		return "dead"
	default:
		return "unknown"
	}
}

// Session tracks the state of a single Claude Code session.
type Session struct {
	mu sync.RWMutex

	ID        string
	Project   string // project directory name
	CWD       string
	GitBranch string
	Version   string
	Model     string

	Status    SessionStatus
	StartedAt time.Time
	UpdatedAt time.Time

	// Metrics
	TotalInputTokens  int
	TotalOutputTokens int
	TotalCacheRead    int
	TotalCacheWrite   int
	ToolCounts        map[string]int

	// Subagents
	Subagents []*Subagent

	// Last event info
	LastToolName string
	LastText     string
}

// Subagent tracks a subagent within a session.
type Subagent struct {
	ID          string
	Type        string // "general-purpose", "Explore", etc.
	Description string
	Status      SessionStatus
	StartedAt   time.Time
	UpdatedAt   time.Time
}

// NewSession creates a new session with default values.
func NewSession(id, project string) *Session {
	now := time.Now()
	return &Session{
		ID:         id,
		Project:    project,
		Status:     StatusActive,
		StartedAt:  now,
		UpdatedAt:  now,
		ToolCounts: make(map[string]int),
	}
}

// UpdateFromEvent updates session state based on an incoming event.
func (s *Session) UpdateFromEvent(e Event) {
	s.mu.Lock()
	defer s.mu.Unlock()

	s.UpdatedAt = e.Timestamp

	if e.Model != "" {
		s.Model = e.Model
	}

	switch e.Type {
	case EventToolUse:
		s.Status = StatusActive
		s.LastToolName = e.ToolName
		s.ToolCounts[e.ToolName]++
	case EventAssistantText:
		s.Status = StatusActive
		s.TotalInputTokens += e.InputTokens
		s.TotalOutputTokens += e.OutputTokens
		s.TotalCacheRead += e.CacheReadInputTokens
		s.TotalCacheWrite += e.CacheCreationInputTokens
		if e.Text != "" {
			s.LastText = e.Text
		}
	case EventUserMessage:
		s.Status = StatusActive
	case EventError:
		s.Status = StatusError
	}
}
```

package data

// RingBuffer is a fixed-capacity circular buffer of Events.
// When full, new pushes overwrite the oldest entry.
type RingBuffer struct {
	buf   []Event
	cap   int
	head  int // next write position
	count int
}

// NewRingBuffer creates a ring buffer with the given capacity.
func NewRingBuffer(capacity int) *RingBuffer {
	return &RingBuffer{
		buf: make([]Event, capacity),
		cap: capacity,
	}
}

// Push adds an event to the buffer, overwriting the oldest if full.
func (rb *RingBuffer) Push(e Event) {
	rb.buf[rb.head] = e
	rb.head = (rb.head + 1) % rb.cap
	if rb.count < rb.cap {
		rb.count++
	}
}

// Len returns the number of events in the buffer.
func (rb *RingBuffer) Len() int {
	return rb.count
}

// All returns all events in order from oldest to newest.
func (rb *RingBuffer) All() []Event {
	return rb.Slice(0, rb.count)
}

// Slice returns events from index start (inclusive) to end (exclusive),
// in insertion order (oldest first).
func (rb *RingBuffer) Slice(start, end int) []Event {
	if start < 0 {
		start = 0
	}
	if end > rb.count {
		end = rb.count
	}
	if start >= end {
		return nil
	}

	result := make([]Event, 0, end-start)
	// The oldest item is at (head - count) mod cap
	oldest := (rb.head - rb.count + rb.cap) % rb.cap
	for i := start; i < end; i++ {
		idx := (oldest + i) % rb.cap
		result = append(result, rb.buf[idx])
	}
	return result
}

// Last returns the most recent n events (newest last).
func (rb *RingBuffer) Last(n int) []Event {
	if n > rb.count {
		n = rb.count
	}
	return rb.Slice(rb.count-n, rb.count)
}
```

package source

import (
	"bytes"
	"encoding/json"
	"fmt"
	"time"

	"github.com/pattynextdoor/toph/internal/data"
)

// rawRecord is the minimal structure we decode from every JSONL line
// to determine the record type before parsing type-specific fields.
type rawRecord struct {
	Type      string          `json:"type"`
	UUID      string          `json:"uuid"`
	Timestamp string          `json:"timestamp"`
	SessionID string          `json:"sessionId"`
	CWD       string          `json:"cwd"`
	GitBranch string          `json:"gitBranch"`
	Version   string          `json:"version"`
	AgentID   string          `json:"agentId"`
	Message   json.RawMessage `json:"message"`
}

type assistantMessage struct {
	Model      string            `json:"model"`
	Content    []json.RawMessage `json:"content"`
	StopReason string            `json:"stop_reason"`
	Usage      struct {
		InputTokens              int `json:"input_tokens"`
		OutputTokens             int `json:"output_tokens"`
		CacheCreationInputTokens int `json:"cache_creation_input_tokens"`
		CacheReadInputTokens     int `json:"cache_read_input_tokens"`
	} `json:"usage"`
}

type contentBlock struct {
	Type  string          `json:"type"`
	Text  string          `json:"text,omitempty"`
	Name  string          `json:"name,omitempty"`
	ID    string          `json:"id,omitempty"`
	Input json.RawMessage `json:"input,omitempty"`
}

type userMessage struct {
	Role    string `json:"role"`
	Content string `json:"content"`
}

// ParseLine parses a single JSONL line and returns zero or more normalized events.
// Returns nil for unknown types or invalid JSON (per error handling spec).
func ParseLine(line []byte, project string) []data.Event {
	line = bytes.TrimSpace(line)
	if len(line) == 0 {
		return nil
	}

	var rec rawRecord
	if err := json.Unmarshal(line, &rec); err != nil {
		return nil
	}

	ts := parseTimestamp(rec.Timestamp)

	switch rec.Type {
	case "user":
		return parseUserRecord(rec, ts, project)
	case "assistant":
		return parseAssistantRecord(rec, ts, project)
	case "system":
		return []data.Event{{
			Type:      data.EventSystemMessage,
			Timestamp: ts,
			SessionID: rec.SessionID,
		}}
	default:
		// Unknown types (file-history-snapshot, progress, queue-operation, last-prompt)
		// are silently skipped per spec.
		return nil
	}
}

// ParseBytes parses an entire JSONL byte slice (multiple lines).
func ParseBytes(b []byte, project string) []data.Event {
	var all []data.Event
	for _, line := range bytes.Split(b, []byte("\n")) {
		if events := ParseLine(line, project); len(events) > 0 {
			all = append(all, events...)
		}
	}
	return all
}

func parseUserRecord(rec rawRecord, ts time.Time, project string) []data.Event {
	var msg userMessage
	if rec.Message != nil {
		json.Unmarshal(rec.Message, &msg)
	}

	text := msg.Content
	if len(text) > 100 {
		text = text[:100] + "..."
	}

	return []data.Event{{
		Type:      data.EventUserMessage,
		Timestamp: ts,
		SessionID: rec.SessionID,
		AgentID:   rec.AgentID,
		Text:      text,
	}}
}

func parseAssistantRecord(rec rawRecord, ts time.Time, project string) []data.Event {
	if rec.Message == nil {
		return nil
	}

	var msg assistantMessage
	if err := json.Unmarshal(rec.Message, &msg); err != nil {
		return nil
	}

	var events []data.Event

	// Parse content blocks for text and tool_use
	var textParts []string
	for _, raw := range msg.Content {
		var block contentBlock
		if err := json.Unmarshal(raw, &block); err != nil {
			continue
		}

		switch block.Type {
		case "text":
			if block.Text != "" {
				summary := block.Text
				if len(summary) > 100 {
					summary = summary[:100] + "..."
				}
				textParts = append(textParts, summary)
			}
		case "tool_use":
			toolEvent := data.Event{
				Type:      data.EventToolUse,
				Timestamp: ts,
				SessionID: rec.SessionID,
				AgentID:   rec.AgentID,
				ToolName:  block.Name,
				Model:     msg.Model,
			}
			// Summarize tool input
			if len(block.Input) > 0 {
				toolEvent.ToolInput = summarizeToolInput(block.Name, block.Input)
			}
			toolEvent.Text = fmt.Sprintf("%s", block.Name)
			events = append(events, toolEvent)
		}
	}

	// Create the main assistant text event with token usage
	textSummary := ""
	if len(textParts) > 0 {
		textSummary = textParts[0]
	}

	assistantEvent := data.Event{
		Type:                     data.EventAssistantText,
		Timestamp:                ts,
		SessionID:                rec.SessionID,
		AgentID:                  rec.AgentID,
		Text:                     textSummary,
		Model:                    msg.Model,
		StopReason:               msg.StopReason,
		InputTokens:              msg.Usage.InputTokens,
		OutputTokens:             msg.Usage.OutputTokens,
		CacheCreationInputTokens: msg.Usage.CacheCreationInputTokens,
		CacheReadInputTokens:     msg.Usage.CacheReadInputTokens,
	}
	events = append(events, assistantEvent)

	return events
}

// summarizeToolInput produces a short description of what a tool is doing.
func summarizeToolInput(toolName string, input json.RawMessage) string {
	var m map[string]interface{}
	if err := json.Unmarshal(input, &m); err != nil {
		return ""
	}

	switch toolName {
	case "Bash":
		if cmd, ok := m["command"].(string); ok {
			if len(cmd) > 60 {
				cmd = cmd[:60] + "..."
			}
			return cmd
		}
	case "Read":
		if fp, ok := m["file_path"].(string); ok {
			return fp
		}
	case "Write":
		if fp, ok := m["file_path"].(string); ok {
			return fp
		}
	case "Edit":
		if fp, ok := m["file_path"].(string); ok {
			return fp
		}
	case "Glob":
		if p, ok := m["pattern"].(string); ok {
			return p
		}
	case "Grep":
		if p, ok := m["pattern"].(string); ok {
			return p
		}
	}
	return ""
}

func parseTimestamp(s string) time.Time {
	t, err := time.Parse(time.RFC3339Nano, s)

		t, err = time.Parse("2006-01-02T15:04:05.000Z", s)
		if err != nil {

package source

import (
	"context"
	"io"
	"log/slog"
	"os"
	"path/filepath"
	"strings"
	"sync"

	"github.com/fsnotify/fsnotify"

// JSONLSource watches ~/.claude/projects/ for JSONL file changes
// and emits parsed events.
type JSONLSource struct {
	baseDir string
	watcher *fsnotify.Watcher
	mu      sync.Mutex
	offsets map[string]int64 // file path -> last read offset
}

// NewJSONLSource creates a source that watches the given base directory
// for JSONL file changes.
func NewJSONLSource(baseDir string) *JSONLSource {
	return &JSONLSource{
		baseDir: baseDir,
		offsets: make(map[string]int64),
	}
}

func (s *JSONLSource) Name() string { return "jsonl" }

func (s *JSONLSource) Start(ctx context.Context, events chan<- data.Event) error {
	var err error
	s.watcher, err = fsnotify.NewWatcher()
	if err != nil {
		return err
	}

	// Discover and watch existing project dirs + JSONL files
	s.discoverAndWatch()

	// Backfill: read tail of existing JSONL files
	s.backfill(events)

	// Watch loop
	for {
		select {
		case <-ctx.Done():
			return nil
		case event, ok := <-s.watcher.Events:
			if !ok {
				return nil
			}
			s.handleFSEvent(event, events)
		case err, ok := <-s.watcher.Errors:
			if !ok {
				return nil
			}
			slog.Debug("fsnotify error", "error", err)
		}
	}
}

func (s *JSONLSource) Stop() error {
	if s.watcher != nil {
		return s.watcher.Close()
	}
	return nil
}

// discoverAndWatch adds watches on all project dirs and discovers JSONL files.
func (s *JSONLSource) discoverAndWatch() {
	entries, err := os.ReadDir(s.baseDir)
	if err != nil {
		slog.Debug("cannot read base dir", "path", s.baseDir, "error", err)
		return
	}

	// Watch the base dir for new project dirs
	s.watcher.Add(s.baseDir)

	for _, entry := range entries {
		if !entry.IsDir() {
			continue
		}
		projDir := filepath.Join(s.baseDir, entry.Name())
		s.watcher.Add(projDir)

		// Also watch subagent dirs
		subDir := filepath.Join(projDir, "*", "subagents")
		matches, _ := filepath.Glob(subDir)
		for _, m := range matches {
			s.watcher.Add(m)
		}

		// Watch session dirs (for subagent creation)
		sessionDirs, _ := filepath.Glob(filepath.Join(projDir, "*"))
		for _, sd := range sessionDirs {
			info, err := os.Stat(sd)
			if err == nil && info.IsDir() {
				s.watcher.Add(sd)
			}
		}
	}
}

// backfill reads the last ~200 lines of each existing JSONL file.
func (s *JSONLSource) backfill(events chan<- data.Event) {
	matches, _ := filepath.Glob(filepath.Join(s.baseDir, "*", "*.jsonl"))
	subMatches, _ := filepath.Glob(filepath.Join(s.baseDir, "*", "*", "subagents", "*.jsonl"))
	matches = append(matches, subMatches...)

	for _, path := range matches {
		s.readTail(path, 200, events)
	}
}

// readTail reads the last n lines of a JSONL file and emits events.
func (s *JSONLSource) readTail(path string, maxLines int, events chan<- data.Event) {
	f, err := os.Open(path)
	if err != nil {
		return
	}
	defer f.Close()

	project := s.projectFromPath(path)

	content, err := io.ReadAll(f)
	if err != nil {
		return
	}

	lines := strings.Split(string(content), "\n")

	// Take last maxLines
	start := 0
	if len(lines) > maxLines {
		start = len(lines) - maxLines
	}

	for _, line := range lines[start:] {
		if line == "" {
			continue
		}
		for _, e := range ParseLine([]byte(line), project) {
			events <- e
		}
	}

	// Record offset for subsequent reads
	s.mu.Lock()
	s.offsets[path] = int64(len(content))
	s.mu.Unlock()
}

// handleFSEvent processes fsnotify events.
func (s *JSONLSource) handleFSEvent(event fsnotify.Event, events chan<- data.Event) {
	// New directory: add to watch
	if event.Has(fsnotify.Create) {
		info, err := os.Stat(event.Name)
		if err == nil && info.IsDir() {
			s.watcher.Add(event.Name)
			return
		}
	}

	// Only care about writes to .jsonl files
	if !event.Has(fsnotify.Write) {
		return
	}
	if !strings.HasSuffix(event.Name, ".jsonl") {
		return
	}

	s.readNewLines(event.Name, events)
}

// readNewLines reads bytes appended since last read.
func (s *JSONLSource) readNewLines(path string, events chan<- data.Event) {
	s.mu.Lock()
	offset := s.offsets[path]
	s.mu.Unlock()

	f, err := os.Open(path)
	if err != nil {
		return
	}
	defer f.Close()

	// Seek to last known position
	if offset > 0 {
		f.Seek(offset, io.SeekStart)
	}

	newContent, err := io.ReadAll(f)
	if err != nil || len(newContent) == 0 {
		return
	}

	project := s.projectFromPath(path)

	for _, line := range strings.Split(string(newContent), "\n") {
		if line == "" {
			continue
		}
		for _, e := range ParseLine([]byte(line), project) {
			events <- e
		}
	}

	s.mu.Lock()
	s.offsets[path] = offset + int64(len(newContent))
	s.mu.Unlock()
}

// projectFromPath extracts the project name from a JSONL file path.
func (s *JSONLSource) projectFromPath(path string) string {
	// path: baseDir/project-name/session.jsonl
	// or:   baseDir/project-name/session-uuid/subagents/agent.jsonl
	rel, err := filepath.Rel(s.baseDir, path)
	if err != nil {
		return "unknown"
	}
	parts := strings.SplitN(rel, string(filepath.Separator), 2)
	if len(parts) > 0 {
		return parts[0]
	}
	return "unknown"
}

const (
	// ActivityBufferSize is the max events in the activity feed.
	ActivityBufferSize = 1000

	// DefaultActiveThreshold is how recently a session must have been
	// updated to be considered "active".
	DefaultActiveThreshold = 5 * time.Minute

// Manager owns all session state and the global activity feed.
type Manager struct {
	mu       sync.RWMutex
	sessions map[string]*Session
	feed     *RingBuffer
}

// NewManager creates a new session manager.
func NewManager() *Manager {
	return &Manager{
		sessions: make(map[string]*Session),
		feed:     NewRingBuffer(ActivityBufferSize),
	}
}

// HandleEvent processes an incoming event: updates session state
// and appends to the activity feed.
func (m *Manager) HandleEvent(e Event) {
	m.mu.Lock()
	defer m.mu.Unlock()

	// Get or create session
	sess, ok := m.sessions[e.SessionID]
	if !ok {
		sess = NewSession(e.SessionID, "")
		m.sessions[e.SessionID] = sess
	}

	sess.UpdateFromEvent(e)
	m.feed.Push(e)
}

// Sessions returns all known sessions sorted by last update (newest first).
func (m *Manager) Sessions() []*Session {
	m.mu.RLock()
	defer m.mu.RUnlock()

	result := make([]*Session, 0, len(m.sessions))
	for _, s := range m.sessions {
		result = append(result, s)
	}

	sort.Slice(result, func(i, j int) bool {
		return result[i].UpdatedAt.After(result[j].UpdatedAt)
	})

	return result
}

// ActiveSessions returns sessions updated within the given threshold.
func (m *Manager) ActiveSessions(threshold time.Duration) []*Session {
	m.mu.RLock()
	defer m.mu.RUnlock()

	cutoff := time.Now().Add(-threshold)
	var result []*Session
	for _, s := range m.sessions {
		if s.UpdatedAt.After(cutoff) {
			result = append(result, s)
		}
	}

	sort.Slice(result, func(i, j int) bool {
		return result[i].UpdatedAt.After(result[j].UpdatedAt)
	})

	return result
}

// ActivityFeed returns all events in the activity feed (oldest first).
func (m *Manager) ActivityFeed() []Event {
	m.mu.RLock()
	defer m.mu.RUnlock()
	return m.feed.All()
}

// ActivityFeedLast returns the last n events.
func (m *Manager) ActivityFeedLast(n int) []Event {
	m.mu.RLock()
	defer m.mu.RUnlock()
	return m.feed.Last(n)
}

// SessionByID returns a session by ID, or nil if not found.
func (m *Manager) SessionByID(id string) *Session {
	m.mu.RLock()
	defer m.mu.RUnlock()
	return m.sessions[id]
}

// ToolCounts returns aggregated tool usage across all sessions.
func (m *Manager) ToolCounts() map[string]int {
	m.mu.RLock()
	defer m.mu.RUnlock()

	totals := make(map[string]int)
	for _, s := range m.sessions {
		s.mu.RLock()
		for tool, count := range s.ToolCounts {
			totals[tool] += count
		}
		s.mu.RUnlock()
	}
	return totals
}

package ui

// Layout holds computed panel dimensions for the current terminal size.
type Layout struct {
	Width  int
	Height int

	// Column widths
	LeftWidth  int
	RightWidth int

	// Row heights (left column)
	SessionsHeight int
	DetailHeight   int
	ToolsHeight    int

	// Row heights (right column)
	ActivityHeight int
	MetricsHeight  int

	// Status bar
	StatusBarHeight int

	// Minimum viable size
	TooSmall bool
}

const (
	minWidth  = 60
	minHeight = 20

	// Border + padding overhead per panel (rounded border = 2 chars each side)
	panelOverheadH = 4 // left border + left pad + right pad + right border
	panelOverheadV = 2 // top border + bottom border
)

// ComputeLayout calculates panel sizes for the given terminal dimensions.
func ComputeLayout(width, height int) Layout {
	l := Layout{
		Width:           width,
		Height:          height,
		StatusBarHeight: 1,
	}

	if width < minWidth || height < minHeight {
		l.TooSmall = true
		return l
	}

	usableHeight := height - l.StatusBarHeight

	// Left column: 35% of width, right column: 65%
	l.LeftWidth = width * 35 / 100
	l.RightWidth = width - l.LeftWidth

	// Left column split: sessions 40%, detail 40%, tools 20%
	l.SessionsHeight = usableHeight * 40 / 100
	l.ToolsHeight = usableHeight * 20 / 100
	l.DetailHeight = usableHeight - l.SessionsHeight - l.ToolsHeight

	// Right column split: activity 70%, metrics 30%
	l.ActivityHeight = usableHeight * 70 / 100
	l.MetricsHeight = usableHeight - l.ActivityHeight

	return l
}

// InnerWidth returns the usable width inside a panel (minus border/padding).
func (l Layout) InnerWidth(panelWidth int) int {
	w := panelWidth - panelOverheadH
	if w < 0 {
		return 0
	}
	return w
}

// InnerHeight returns the usable height inside a panel (minus border).
func (l Layout) InnerHeight(panelHeight int) int {
	h := panelHeight - panelOverheadV
	if h < 0 {
		return 0
	}
	return h
}

func (p *SessionsPanel) renderSessionRow(s *data.Session, selected bool, width int) string {

	switch s.Status {
	case data.StatusActive:
		icon = "●"
		iconColor = p.theme.Active
	case data.StatusWaiting:
		icon = "◐"
		iconColor = p.theme.Waiting
	case data.StatusIdle:
		icon = "○"
		iconColor = p.theme.Idle
	case data.StatusError:
		icon = "✕"
		iconColor = p.theme.Error
	default:
		icon = "○"
		iconColor = p.theme.Idle
	}

	iconStyle := lipgloss.NewStyle().Foreground(iconColor)

	// Project name
	project := s.Project
	if project == "" {
		project = s.ID[:min(8, len(s.ID))]
	}

	// Branch
	branch := s.GitBranch
	if branch == "" {
		branch = "-"
	}

	// Age
	age := formatDuration(time.Since(s.UpdatedAt))

	// Build row
	row := fmt.Sprintf("%s %s %s %s",
		iconStyle.Render(icon),
		project,
		lipgloss.NewStyle().Foreground(p.theme.TextDim).Render(branch),
		lipgloss.NewStyle().Foreground(p.theme.TextDim).Render(age),
	)

	if selected {
		row = lipgloss.NewStyle().
			Background(lipgloss.Color("#303030")).
			Width(width).
			Render(row)

func formatDuration(d time.Duration) string {

		return fmt.Sprintf("%ds", int(d.Seconds()))
	}
	if d < time.Hour {
		return fmt.Sprintf("%dm", int(d.Minutes()))
	}
	return fmt.Sprintf("%dh%dm", int(d.Hours()), int(d.Minutes())%60)
}

	p.offset = 999999

// Render draws the activity feed panel.
func (p *ActivityPanel) Render(events []data.Event, width, height int) string {
	style := p.theme.PanelNormal
	if p.focused {
		style = p.theme.PanelFocus
	}

	innerW := width - 4
	innerH := height - 2
	if innerW < 1 || innerH < 1 {
		return style.Width(width - 2).Height(height - 2).Render("")
	}

	title := p.theme.Title.Render("ACTIVITY")
	visibleLines := innerH - 1 // -1 for title

	// Clamp offset
	maxOffset := len(events) - visibleLines
	if maxOffset < 0 {
		maxOffset = 0
	}
	if p.offset > maxOffset {
		p.offset = maxOffset
	}

	// Calculate visible range
	end := len(events) - p.offset
	start := end - visibleLines
	if start < 0 {
		start = 0
	}

	var lines []string
	lines = append(lines, title)

	if len(events) == 0 {
		lines = append(lines, lipgloss.NewStyle().
			Foreground(p.theme.Idle).
			Render("Waiting for events..."))
	}

	for _, e := range events[start:end] {
		line := p.renderEvent(e, innerW)
		lines = append(lines, line)
	}

	// Show "new" indicator if scrolled up
	if !p.autoScroll && p.offset > 0 {
		indicator := lipgloss.NewStyle().
			Foreground(p.theme.Waiting).
			Render(fmt.Sprintf("  ↓ %d new", p.offset))
		lines = append(lines, indicator)
	}

	return style.Width(width - 2).Height(height - 2).Render(content)

		typeColor = p.theme.Subagent
	case data.EventError:
		typeLabel = "ERROR"
		typeColor = p.theme.Error
	default:
		typeLabel = e.Type.String()
		typeColor = p.theme.TextDim
	}

	labelStyle := lipgloss.NewStyle().Foreground(typeColor)

	// Session ID (first 6 chars)
	sessID := e.SessionID
	if len(sessID) > 6 {
		sessID = sessID[:6]
	}

	// Build the line
	prefix := fmt.Sprintf("%s %s %-8s",
		tsStyle.Render(ts),
		tsStyle.Render(sessID),

	}
	return prefix

func (p *ToolsPanel) SetFocused(f bool) { p.focused = f }

// Focused returns whether this panel is focused.
func (p *ToolsPanel) Focused() bool { return p.focused }

// Render draws the tool usage bar chart.
func (p *ToolsPanel) Render(toolCounts map[string]int, width, height int) string {
	style := p.theme.PanelNormal
	if p.focused {
		style = p.theme.PanelFocus
	}

	innerW := width - 4
	innerH := height - 2
	if innerW < 1 || innerH < 1 {
		return style.Width(width - 2).Height(height - 2).Render("")
	}

	title := p.theme.Title.Render("TOOLS")

	var lines []string
	lines = append(lines, title)

	if len(toolCounts) == 0 {
		lines = append(lines, lipgloss.NewStyle().
			Foreground(p.theme.Idle).
			Render("No tool calls yet"))
		content := strings.Join(lines, "\n")
		return style.Width(width - 2).Height(height - 2).Render(content)
	}

	// Sort by count descending
	type toolEntry struct {
		name  string
		count int
	}
	var entries []toolEntry
	for name, count := range toolCounts {
		entries = append(entries, toolEntry{name, count})
	}
	sort.Slice(entries, func(i, j int) bool {
		return entries[i].count > entries[j].count
	})

	// Find max for scaling
	maxCount := entries[0].count

	// Label width: longest tool name + count
	labelWidth := 0
	for _, e := range entries {
		l := len(fmt.Sprintf("%-6s %3d ", e.name, e.count))
		if l > labelWidth {
			labelWidth = l
		}
	}

	barWidth := innerW - labelWidth - 1
	if barWidth < 5 {
		barWidth = 5
	}

	maxRows := innerH - 1 // -1 for title
	for i, e := range entries {
		if i >= maxRows {
			break
		}

		barLen := e.count * barWidth / maxCount
		if barLen < 1 {
			barLen = 1
		}

		bar := strings.Repeat("█", barLen)
		label := fmt.Sprintf("%-6s %3d ", e.name, e.count)

		lines = append(lines, fmt.Sprintf("%s%s",
			label,
			lipgloss.NewStyle().Foreground(p.theme.ToolUse).Render(bar),
		))
	}

	content := strings.Join(lines, "\n")
	return style.Width(width - 2).Height(height - 2).Render(content)
}

package model

import (
	"time"

	tea "github.com/charmbracelet/bubbletea/v2"
	"github.com/charmbracelet/lipgloss/v2"
	"github.com/pattynextdoor/toph/internal/data"
	"github.com/pattynextdoor/toph/internal/ui"
	"github.com/pattynextdoor/toph/internal/ui/panels"
)

const tickInterval = 33 * time.Millisecond // ~30fps

// tickMsg is sent every 33ms to trigger a render.
type tickMsg time.Time

// eventMsg wraps incoming data events from sources.
type EventMsg data.Event

// Model is the root Bubble Tea model for toph.
type Model struct {
	// Data
	manager *data.Manager

	// UI
	theme     *ui.Theme
	layout    ui.Layout
	statusBar *ui.StatusBar

	// Panels
	sessions *panels.SessionsPanel
	activity *panels.ActivityPanel
	tools    *panels.ToolsPanel

	// State
	focusedPanel int
	width        int
	height       int
	ready        bool

	// Pending events to batch between ticks
	pendingEvents []data.Event
}

// New creates a new root model.
func New(manager *data.Manager) Model {
	theme := ui.DefaultTheme()
	return Model{
		manager:   manager,
		theme:     theme,
		statusBar: ui.NewStatusBar(theme),
		sessions:  panels.NewSessionsPanel(theme),
		activity:  panels.NewActivityPanel(theme),
		tools:     panels.NewToolsPanel(theme),
	}
}

func (m Model) Init() tea.Cmd {
	return tick()
}

func (m Model) Update(msg tea.Msg) (tea.Model, tea.Cmd) {
	switch msg := msg.(type) {
	case tea.WindowSizeMsg:
		m.width = msg.Width
		m.height = msg.Height
		m.layout = ui.ComputeLayout(m.width, m.height)
		m.ready = true
		return m, nil

	case tickMsg:
		// Flush pending events
		for _, e := range m.pendingEvents {
			m.manager.HandleEvent(e)
		}
		m.pendingEvents = m.pendingEvents[:0]
		return m, tick()

	case EventMsg:
		// Buffer events for next tick
		m.pendingEvents = append(m.pendingEvents, data.Event(msg))
		return m, nil

	case tea.KeyPressMsg:
		return m.handleKey(msg)
	}

	return m, nil
}

func (m Model) handleKey(msg tea.KeyPressMsg) (tea.Model, tea.Cmd) {
	if isQuit(msg) {
		return m, tea.Quit
	}

	if panel, ok := isJumpToPanel(msg); ok {
		m.setFocus(panel)
		return m, nil
	}

	if isTab(msg) {
		m.setFocus((m.focusedPanel + 1) % PanelCount)
		return m, nil
	}
	if isShiftTab(msg) {
		m.setFocus((m.focusedPanel - 1 + PanelCount) % PanelCount)
		return m, nil
	}

	// Panel-specific keys
	switch m.focusedPanel {
	case PanelSessions:
		if isUp(msg) {
			m.sessions.MoveUp()
		} else if isDown(msg) {
			m.sessions.MoveDown(len(m.manager.ActiveSessions(data.DefaultActiveThreshold)))
		}
	case PanelActivity:
		switch {
		case isUp(msg):
			m.activity.ScrollUp(1)
		case isDown(msg):
			m.activity.ScrollDown(1)
		case msg.String() == "G":
			m.activity.ScrollToBottom()
		case msg.String() == "g":
			m.activity.ScrollToTop()
		}
	}

	return m, nil
}

func (m *Model) setFocus(panel int) {
	m.focusedPanel = panel
	m.sessions.SetFocused(panel == PanelSessions)
	m.activity.SetFocused(panel == PanelActivity)
	m.tools.SetFocused(panel == PanelTools)
}

func (m Model) View() string {
	if !m.ready {
		return "Initializing toph..."
	}

	if m.layout.TooSmall {
		return lipgloss.Place(m.width, m.height,
			lipgloss.Center, lipgloss.Center,
			"Terminal too small (min 60x20)")
	}

	l := m.layout

	// Get data
	activeSessions := m.manager.ActiveSessions(data.DefaultActiveThreshold)
	allSessions := m.manager.Sessions()
	feedEvents := m.manager.ActivityFeed()
	toolCounts := m.manager.ToolCounts()

	// Render panels
	sessionsView := m.sessions.Render(allSessions, l.LeftWidth, l.SessionsHeight)
	activityView := m.activity.Render(feedEvents, l.RightWidth, l.ActivityHeight)
	toolsView := m.tools.Render(toolCounts, l.LeftWidth, l.ToolsHeight)

	// Detail panel placeholder (Phase 2)
	detailStyle := m.theme.PanelNormal
	if m.focusedPanel == PanelDetail {
		detailStyle = m.theme.PanelFocus
	}
	detailView := detailStyle.
		Width(l.LeftWidth - 2).
		Height(l.DetailHeight - 2).
		Render(m.theme.Title.Render("DETAIL") + "\n" +
			lipgloss.NewStyle().Foreground(m.theme.Idle).Render("Select a session"))

	// Metrics panel placeholder (Phase 2)
	metricsStyle := m.theme.PanelNormal
	if m.focusedPanel == PanelMetrics {
		metricsStyle = m.theme.PanelFocus
	}
	metricsView := metricsStyle.
		Width(l.RightWidth - 2).
		Height(l.MetricsHeight - 2).
		Render(m.theme.Title.Render("METRICS") + "\n" +
			lipgloss.NewStyle().Foreground(m.theme.Idle).Render("Collecting..."))

	// Compose layout
	leftCol := lipgloss.JoinVertical(lipgloss.Left,
		sessionsView,
		detailView,
		toolsView,
	)

	rightCol := lipgloss.JoinVertical(lipgloss.Left,
		activityView,
		metricsView,
	)

	mainView := lipgloss.JoinHorizontal(lipgloss.Top,
		leftCol,
		rightCol,
	)

	statusBar := m.statusBar.Render(m.width, len(activeSessions), "jsonl", true)

	return lipgloss.JoinVertical(lipgloss.Left, mainView, statusBar)
}

func tick() tea.Cmd {
	return tea.Tick(tickInterval, func(t time.Time) tea.Msg {
		return tickMsg(t)
	})
}

package main

import (
	"context"
	"fmt"
	"os"
	"os/signal"
	"path/filepath"
	"syscall"

	tea "github.com/charmbracelet/bubbletea/v2"
	"github.com/pattynextdoor/toph/internal/data"
	"github.com/pattynextdoor/toph/internal/model"
	"github.com/pattynextdoor/toph/internal/source"
)

func main() {
	// Resolve Claude projects directory
	home, err := os.UserHomeDir()
	if err != nil {
		fmt.Fprintf(os.Stderr, "Error: cannot find home directory: %v\n", err)
		os.Exit(1)
	}
	projectsDir := filepath.Join(home, ".claude", "projects")

	// Check if projects dir exists
	if _, err := os.Stat(projectsDir); os.IsNotExist(err) {
		fmt.Fprintf(os.Stderr, "No Claude Code data found at %s\n", projectsDir)
		fmt.Fprintf(os.Stderr, "Run Claude Code at least once to generate session data.\n")
		os.Exit(1)
	}

	// Context with signal handling
	ctx, cancel := context.WithCancel(context.Background())
	defer cancel()

	sigCh := make(chan os.Signal, 1)
	signal.Notify(sigCh, syscall.SIGINT, syscall.SIGTERM)
	go func() {
		<-sigCh
		cancel()
	}()

	// Create session manager
	manager := data.NewManager()

	// Create JSONL source
	jsonlSource := source.NewJSONLSource(projectsDir)

	// Event channel bridges source → Bubble Tea
	eventCh := make(chan data.Event, 256)

	// Start source in background
	go func() {
		if err := jsonlSource.Start(ctx, eventCh); err != nil {
			fmt.Fprintf(os.Stderr, "Source error: %v\n", err)
		}
	}()

	// Create Bubble Tea program
	m := model.New(manager)
	p := tea.NewProgram(m, tea.WithAltScreen())

	// Bridge: forward events from source channel to Bubble Tea messages
	go func() {
		for {
			select {
			case <-ctx.Done():
				return
			case e, ok := <-eventCh:
				if !ok {
					return
				}
				p.Send(model.EventMsg(e))
			}
		}
	}()

	// Run TUI (blocks until quit)
	if _, err := p.Run(); err != nil {
		fmt.Fprintf(os.Stderr, "Error: %v\n", err)
		os.Exit(1)
	}

	// Cleanup
	cancel()
	jsonlSource.Stop()
}
